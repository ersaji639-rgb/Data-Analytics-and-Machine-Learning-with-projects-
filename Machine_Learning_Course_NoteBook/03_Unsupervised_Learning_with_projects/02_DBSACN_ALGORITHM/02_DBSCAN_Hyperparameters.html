def display_category(model, data):
    labels = model.fit_predict(data)
    sns.scatterplot(data = data, x = 'X1', y ='X2', hue = labels, palette = 'Set1')

outlier_percent = []
no_of_outlier = []
for ep in np.linspace(0.001,10,100):
    # create model 
    dbscan = DBSCAN(eps= ep)
    dbscan.fit(two_blobs_outliers)

    no_of_outlier.append(np.sum(dbscan.labels_ == -1)) # no of outliers 

    # Percentage of points that are clusters
    outlier_percent.append(100*np.sum(dbscan.labels_ == -1)/len(dbscan.labels_))
